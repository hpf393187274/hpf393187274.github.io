¶ Confirm 用法

data 数据 border 边框

<template>
  <me-button @click="onHandler">你叫什么名字</me-button>
</template>
<script>
export default {
  methods: {
    onHandler() {
      this.$dialog.confirm({
        content: '请确认：你叫什么名字',
        ok() {
          this.$dialog.alert({
            content: '点击了确认'
          })
        },
        cancel() {
          this.$dialog.alert({
            content: '点击了取消'
          })
        }
      })
    }
  }
}
</script>

¶ Prompt 用法

data 数据 border 边框

<template>
  <me-button @click="onHandler">请输入</me-button>
</template>
<script>
export default {
  methods: {
    onHandler() {
      this.$dialog.prompt({
        content: '请输入文件夹名称',
        rules:[ { required: true, message: '文件夹名称不能为空' }],
        ok(value) {
          this.$message.info(`输入的内容为：${value}`)
        },
        cancel() {
          this.$message.error('点击了取消')
        }
      })
    }
  }
}
</script>
